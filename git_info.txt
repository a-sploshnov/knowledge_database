git init - гит "разворачивается" в текущей папке
git add - добавить файлы в индекс
git diff -  показывает изменения в непроиндексированных файлах
git diff --staged - показывает изменения в проиндексированных файлах
git difftool - просмотр внесенных изменений в графическом просмотре
git commit -m "текст сообщения" - осуществить коммит
git rm --cached имя файла - удалить файл из ожидания
    + file name - удаление файла из индекса
git vm текущее_имя_файла новое_имя_файла - переименование файла
git log - просмотр логов
    -p - показывает разницу внесенных изменений
    --oneline  - логи коротко в одну строку
    --stat - сокращенная статистика для каждого коммита
git commit --amend - переделывает (замена) последний коммит, внося необходимые изменения.Так же можно переименовать название коммита и комментарий.
git reset HEAD + имя файла - удаление файла из индекса
    reset + id "лога" --hard - удаление коммитов которые были после данного коммита
    reset --hard HEAD~число(n) - удаление последних n коммитов
    reset --soft HEAD~число(n) - удаление истории коммитов
git restore --staget + имя файла - замена файла последней зафиксированной версией
git checkout -- имя файла - восстановить файл
git checkout id "лога" - просмотр версии данного лога
git revert + id "лога" - отмена коммита
git branch + имя ветки - создание новой ветки
git checkout master - просмотр текущей версии (переход на ветку)
git checkout -d + имя - создание ветки и переход на неё
git branch -a - просмотор всех веток + узнать на какой сейчас ветке находишься
git merge + имя ветки - объединение веток: текущая + указанная в команде
git branch -d + имя ветки - удаление ветки (возможно только после соединения веток)
git branch -D + имя ветки - удаление ветки
git remote - вывод названия доступных удаленных репозиториев 
    -v - просмотр адресов для чтения и записи, привязанные к репозиторию
    show имя_репозитория - получить больше информации об удоленном репозитории
    rename текущее_имя_репозитория + новое_имя_репозитория - переименовать удаленный репозиторий
    rm - удаление удаленного репозитория
    add имя_репозитория + URL-адрес - подключение к репозиторию
git fetch имя_репозитория - узнать какие изменения есть в удаленном репозитории
git pull - скопировать из репозитория новые файлы (для работы более 1 чел)
git push -u имя_репозитория + имя ветки - загрузка проекта из локальной версии в репозиторий
git clone + URL-адрес - скачивание проекта из репозитория в текущий каталог
git checkout master - просмотр текущей версии (переход на ветку)
git revert + id "лога" - отмена коммита
git reset + id "лога" --hard - удаление коммитов которые были после данного коммита
git branch + имя ветки - создание новой ветки
git branch -a - просмотор всех веток + узнать на какой сейчас ветке находишься
git merge + имя ветки - объединение веток: текущая + указанная в команде
git clone + URL-адрес - скачивание проекта из репозитория в текущий каталог
git pull - скопировать из репозитория новые файлы (для работы более 1 чел)
:wq - выход из "странного" режима командной строки
